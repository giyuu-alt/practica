<?php
$servername = "localhost";
$username = "root";
$password = "";
$dbname = "base_imc";

// Crear conexión
$conn = new mysqli($servername, $username, $password, $dbname);

// Verificar conexión
if ($conn->connect_error) {
    die("Conexión fallida: " . $conn->connect_error);
}

if ($_SERVER['REQUEST_METHOD'] == 'POST') {
    $peso = $_POST['peso'];
    $altura = $_POST['altura'] / 100;  // Convertir altura de cm a metros
    $imc = $peso / ($altura * $altura);  

    // Obtener la última ID de persona insertada
    $sql = "SELECT MAX(id_persona) AS max_id FROM persona";
    $result = $conn->query($sql);
    $row = $result->fetch_assoc();
    $id_persona1 = $row['max_id']; // Asignar la ID más alta a id_persona1

    // Insertar los datos en la tabla 'imc'
    $sql = "INSERT INTO imc (peso, altura, cal_imc, id_persona1) 
            VALUES ('$peso', '$altura', '$imc', '$id_persona1')";

    if ($conn->query($sql) === TRUE) {
        echo "<script>alert('Datos guardados correctamente.');</script>";
        echo "<script>window.location.href='bienvenido.php';</script>";
    } else {
        echo "Error al guardar los datos: " . $conn->error;
    }
}

$conn->close();
